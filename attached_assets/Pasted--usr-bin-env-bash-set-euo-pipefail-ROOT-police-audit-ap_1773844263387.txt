# SQL schema
mkdir -p "$ROOT/sql"
cat > "$ROOT/sql/schema.sql" <<'SQL'
-- PostgreSQL schema (initial)
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  name TEXT,
  email TEXT UNIQUE,
  created_at TIMESTAMP DEFAULT now()
);

CREATE TABLE incidents (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES users(id),
  device_hash TEXT,
  media_url TEXT,
  location_geo JSONB,
  created_at TIMESTAMP DEFAULT now(),
  analysis_status TEXT,
  report_id UUID
);

CREATE TABLE events (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  incident_id UUID REFERENCES incidents(id),
  start_ts FLOAT,
  end_ts FLOAT,
  type TEXT,
  confidence FLOAT,
  transcript_segment TEXT
);

CREATE TABLE officers (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  name TEXT,
  badge_no TEXT,
  agency TEXT,
  roster_url TEXT,
  public_records JSONB,
  flags_summary JSONB
);

CREATE TABLE audit_logs (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  object_id UUID,
  action TEXT,
  actor_id UUID,
  timestamp TIMESTAMP DEFAULT now(),
  signature TEXT
);
SQL
